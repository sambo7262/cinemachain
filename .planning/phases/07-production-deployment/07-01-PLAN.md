The final compose.yaml should have this structure:
```yaml
# compose.yaml — CinemaChain
# [header comments as above]

services:
  backend:
    image: sambo7262/cinemachain-backend:latest
    container_name: cinemachain-backend
    restart: unless-stopped
    mem_limit: 512m
    networks:
      - cinemachain-net
    ports:
      - "8111:8000"
    environment:
      - PUID=${PUID}
      - PGID=${PGID}
      - DATABASE_URL=postgresql+asyncpg://cinema:${DB_PASSWORD}@postgres:5432/cinemachain
      - TMDB_API_KEY=${TMDB_API_KEY}
      - RADARR_URL=${RADARR_URL}
      - RADARR_API_KEY=${RADARR_API_KEY}
      - RADARR_QUALITY_PROFILE=${RADARR_QUALITY_PROFILE:-HD+}
      - TMDB_CACHE_TIME=${TMDB_CACHE_TIME:-03:00}
      - TMDB_CACHE_TOP_N=${TMDB_CACHE_TOP_N:-5000}
      - TMDB_CACHE_RUN_ON_STARTUP=${TMDB_CACHE_RUN_ON_STARTUP:-false}
      - SETTINGS_ENCRYPTION_KEY=${SETTINGS_ENCRYPTION_KEY:-}
    volumes:
      - ${DATA_DIR:-./data}/backend:/app/data
      - ${DATA_DIR:-./data}/static:/app/static
    depends_on:
      postgres:
        condition: service_healthy
    command: >
      bash -c "alembic upgrade head && uvicorn app.main:app --host 0.0.0.0 --port 8000"

  postgres:
    image: postgres:17-alpine
    container_name: cinemachain-postgres
    restart: unless-stopped
    mem_limit: 256m
    networks:
      - cinemachain-net
    ports:
      - "127.0.0.1:5433:5432"
    environment:
      - POSTGRES_USER=cinema
      - POSTGRES_PASSWORD=${DB_PASSWORD}
      - POSTGRES_DB=cinemachain
      - PGDATA=/var/lib/postgresql/data/pgdata
    volumes:
      - ${DATA_DIR:-./data}/postgres:/var/lib/postgresql/data
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U cinema -d cinemachain"]
      interval: 10s
      timeout: 5s
      retries: 5
      start_period: 30s

  frontend:
    image: sambo7262/cinemachain-frontend:latest
    container_name: cinemachain-frontend
    restart: unless-stopped
    mem_limit: 64m
    networks:
      - cinemachain-net
    ports:
      - "3111:80"
    depends_on:
      - backend

networks:
  cinemachain-net:
    driver: bridge
```
