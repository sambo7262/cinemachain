```typescript
// From GameSession.tsx — null-stable two-pass sort
const sorted = [...filtered].sort((a, b) => {
  const getVal = (m: EligibleMovieDTO) => {
    if (sortCol === "rating") return m.vote_average
    if (sortCol === "year")   return m.year
    if (sortCol === "runtime") return m.runtime
    if (sortCol === "rt")     return m.rt_score
    return null
  }
  const aVal = getVal(a)
  const bVal = getVal(b)
  // Nulls always last regardless of direction
  if (aVal == null && bVal == null) return 0
  if (aVal == null) return 1
  if (bVal == null) return -1
  return sortDir === "asc" ? aVal - bVal : bVal - aVal
})
```

Prefix parsing:
```typescript
function parseSearchMode(input: string): { mode: "title" | "person"; query: string } {
  if (input.startsWith("a:") || input.startsWith("d:")) {
    return { mode: "person", query: input.slice(2).trim() }
  }
  if (input.startsWith("m:")) {
    return { mode: "title", query: input.slice(2).trim() }
  }
  return { mode: "title", query: input.trim() }
}
```

```tsx
                <Badge variant="outline">{splashMovie.rt_score}% RT</Badge>
              )}
              {splashMovie.mpaa_rating && <Badge variant="outline">{splashMovie.mpaa_rating}</Badge>}
              {splashMovie.runtime != null && <Badge variant="outline">{splashMovie.runtime}m</Badge>}
              {splashMovie.year && <Badge variant="outline">{splashMovie.year}</Badge>}
            </div>
            <p className="text-sm leading-relaxed text-muted-foreground">
              {splashMovie.overview ?? "No overview available."}
            </p>
            {splashMovie.tmdb_id && (
              <a
                href={`https://www.themoviedb.org/movie/${splashMovie.tmdb_id}`}
                target="_blank"
                rel="noopener noreferrer"
                className="text-sm text-muted-foreground underline mt-2 inline-block"
                aria-label={`View ${splashMovie.title} on TMDB (opens in new tab)`}
              >
                View on TMDB
              </a>
            )}
          </div>
        </div>
        {radarrStatus === "error" && (
          <p className="text-sm text-destructive">
            Request failed. Try again or check Radarr.
          </p>
        )}
        <DialogFooter className="flex gap-2 flex-wrap">
          <Button
            variant="default"
            disabled={radarrLoading}
            onClick={async () => {
              setRadarrLoading(true)
              try {
                const result = await api.requestMovieStandalone(splashMovie.tmdb_id)
                if (result.status === "already_in_radarr") {
                  setRadarrStatus("already_in_radarr")
                } else if (result.status === "queued") {
                  setRadarrStatus("queued")
                } else {
                  setRadarrStatus("error")
                }
              } catch {
                setRadarrStatus("error")
              } finally {
                setRadarrLoading(false)
                if (radarrStatus !== "error") {
                  setTimeout(() => setRadarrStatus(null), 2000)
                }
              }
            }}
          >
            {radarrLoading
              ? "Requesting..."
              : radarrStatus === "queued"
              ? "Added to Radarr"
              : radarrStatus === "already_in_radarr"
              ? "Already in Radarr"
              : "Download via Radarr"}
          </Button>
          <Button
            variant="outline"
            disabled={watchedLoading}
            onClick={async () => {
              setWatchedLoading(true)
              try {
                await api.markWatchedOnline(splashMovie.tmdb_id)
                setRadarrStatus("watched")
              } catch {
                setRadarrStatus("error")
              } finally {
                setWatchedLoading(false)
                setTimeout(() => setRadarrStatus(null), 2000)
              }
            }}
          >
            {watchedLoading
              ? "Saving..."
              : radarrStatus === "watched"
              ? "Marked as Watched"
              : "Watch Online"}
          </Button>
          <Button variant="outline" onClick={() => setSplashOpen(false)}>
            Close
          </Button>
        </DialogFooter>
      </>
    )}
  </DialogContent>
</Dialog>
```
