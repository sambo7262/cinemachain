type SessionStatus = "active" | "paused" | "awaiting_continue" | "ended"

<!-- The 7 UI states the user wants (derived from session status + step count): -->
// no_session        → activeSession is null/undefined
// new_session       → status="active", steps.length === 1 (only starting movie, no actor picked)
// active            → status="active", steps.length > 1 (mid-game, actor picked, picking next movie)
// movie_selected_unwatched → status="active" after requestMovie called (current movie not yet watched)
//   NOTE: backend uses same "active" status for this; differentiate by whether the latest step
//   has a movie_tmdb_id different from the prior step (i.e., movie was just selected)
//   Simplest approach: treat "active" with the most recent step having no actor_name as
//   "movie selected, waiting to be watched" — i.e. last step has movie_tmdb_id but actor_name=null
// awaiting_continue → status="awaiting_continue"
// paused            → status="paused"
// ended             → status="ended" (this state won't appear in active session query, handle in GameSession only)
