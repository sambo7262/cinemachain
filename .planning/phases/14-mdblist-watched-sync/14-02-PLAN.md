2. **_run_watched_sync** — Add async function BELOW `_sync_state`:
   ```python
   async def _run_watched_sync():
       """Background task: push all unsynced WatchEvents to user's MDBList list."""
       try:
           async with _bg_session_factory() as db:
               api_key = await settings_service.get_setting(db, "mdblist_api_key")
               list_id = await settings_service.get_setting(db, "mdblist_list_id")
               if not api_key or not list_id:
                   logger.warning("Watched sync aborted — mdblist_api_key or mdblist_list_id not configured")
                   return

               result = await db.execute(
                   select(WatchEvent).where(WatchEvent.mdblist_synced_at.is_(None))
               )
               events = result.scalars().all()
               _sync_state.total = len(events)
               _sync_state.synced = 0

               if not events:
                   logger.info("Watched sync: nothing to sync")
                   return

               logger.info("Watched sync: starting for %d events", len(events))
               async with httpx.AsyncClient(timeout=10.0) as client:
                   for we in events:
                       if _sync_state.calls_used_today >= _sync_state.daily_limit:
                           logger.warning("Watched sync stopped — daily quota reached (%d)", _sync_state.daily_limit)
                           break

                       try:
                           # Resolve imdb_id from Movie table
                           movie_result = await db.execute(
                               select(Movie.imdb_id).where(Movie.tmdb_id == we.tmdb_id)
                           )
                           imdb_id = movie_result.scalar_one_or_none()

                           # Build identifier payload
                           movie_item: dict = {}
                           if imdb_id:  # truthy — handles None and "" sentinel
                               movie_item["imdb"] = imdb_id
                           else:
                               movie_item["tmdb"] = we.tmdb_id

                           resp = await client.post(
                               f"https://mdblist.com/api/lists/{list_id}/items/add",
                               params={"apikey": api_key},
                               json={"movies": [movie_item]},
                           )
                           if resp.status_code == 429:
                               logger.warning("Watched sync rate limited (429) — waiting 60s")
                               await asyncio.sleep(60)
                               continue
                           if resp.status_code == 200:
                               we.mdblist_synced_at = datetime.utcnow()
                               count = await _increment_quota(db)
                               _sync_state.calls_used_today = count
                               _sync_state.synced += 1
                               await db.commit()
                           else:
                               logger.warning(
                                   "Watched sync: MDBList returned %d for tmdb_id=%d body=%s",
                                   resp.status_code, we.tmdb_id, resp.text[:200],
                               )
                       except Exception:
                           logger.exception("Watched sync: error for tmdb_id=%d", we.tmdb_id)
                           continue

                       await asyncio.sleep(1.0)  # ~1 req/s rate limit

               logger.info("Watched sync complete: %d/%d synced", _sync_state.synced, _sync_state.total)
       finally:
           _sync_state.running = False
   ```

3. **POST /watched-sync/start** — Add endpoint BELOW the existing `/backfill/status` endpoint:
   ```python
   @router.post("/watched-sync/start")
   async def start_watched_sync(background_tasks: BackgroundTasks):
       if _sync_state.running:
           raise HTTPException(409, "Watched sync already running")

       async with _bg_session_factory() as db:
           today_str = date.today().isoformat()
           reset_date = await settings_service.get_setting(db, "mdblist_calls_reset_date")
           if reset_date != today_str:
               _sync_state.calls_used_today = 0
           else:
               count_str = await settings_service.get_setting(db, "mdblist_calls_today") or "0"
               _sync_state.calls_used_today = int(count_str)

           count_result = await db.execute(
               select(func.count(WatchEvent.id)).where(WatchEvent.mdblist_synced_at.is_(None))
           )
           total = count_result.scalar() or 0

       _sync_state.running = True
       _sync_state.synced = 0
       _sync_state.total = total
       background_tasks.add_task(_run_watched_sync)
       return {"started": True, "total": total}
   ```
